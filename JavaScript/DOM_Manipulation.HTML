function rando() {         
    var fruits = ["Bananas", "Oranges", "Apples", "Mangos", "Watermelons", "Dragonfruit",
 "Grapes", "Pears", "Pumpkins", "Elderberries", "Papayas", 
"Salmonberries", "Huckleberries", "Apricots", "Avocado", "Peaches", "Chilies",
 "Honeydews", "Jujubes", "Figs", "Jackfruit", "Nectarines", "Pineapples", "Strawberries", "Limes"];
    var pick = Math.floor(Math.random() * fruits.length);
    return fruits[pick];
}
function append() {
    var node = document.createTextNode(rando() + ", "),
    pre = document.getElementById("domOut");

    pre.append(node);
}
function prepend() {
    var node = document.createTextNode(rando() + ", "),
    pre = document.getElementById("domOut");

    pre.prepend(node);
}
function insert() {
    var para = document.createElement("p");
    var node = document.createTextNode("Your friend found some " + rando() + ". ");
    para.appendChild(node);

    var element = document.getElementById("domDiv");
    var child = document.getElementById("domOut");
    element.insertBefore(para,child);
}   

function create() {
    var para = document.createElement("p");
    var node = document.createTextNode("You found some " + rando() + ".");
    para.append(node);

    var element = document.getElementById("domDiv");
    element.append(para);
}
function remove() {
    var parent = document.getElementById("domOut");
    var child = document.getElementById("pre");
    parent.remove(child);
}
