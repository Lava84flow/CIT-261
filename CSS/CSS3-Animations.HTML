JavaScript Triggers

	 			
<form>
	Push the Button
	<button type="button" onmouseover="spin(this)" 
	onmouseout="resetAnime(this)" onclick="triggerDemo()">
		Push Me
	</button>

function spin(x) {
	x.style.animation = "spin 0.75s ease-in-out";
	x.style.animationIterationCount = "4";
}
function resetAnime(x) {
	x.style.removeProperty('animation');
	x.style.removeProperty('animationIterationCount');
}
function triggerDemo() {
	x = document.getElementById('triggerOut')
	x.style.animation = "rainbow 2s linear";
	x.style.animationIterationCount = "2";
	x.addEventListener("animationend", function() { resetAnime(x); });
}

@keyframes spin {
	0% {}
	100% {transform: rotate(360deg);}
}
