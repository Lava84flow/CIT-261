function textColorDemo(x) {
    if (x.style.color == "orange" && x.style.backgroundColor == "green") {
            x.style.removeProperty('color');
            x.style.removeProperty('background-color');                    
        } else {
            x.style.color = "orange";
            x.style.backgroundColor = "green";
        }            
}

function textSizeDemo(x) {
    if (x.style.fontSize == "40px") {
            x.style.removeProperty('font-size');
        } else {
            x.style.fontSize = "40px";
        }            
}

function marginDemo(x) {
    if (x.style.padding == "75px") {
            x.style.removeProperty('padding');
        } else {
            x.style.padding = "75px";
        }            
}

function backColorDemo(x) {
    if (x.style.backgroundColor == "darkorange") {
            x.style.removeProperty('background-color');
        } else {
            x.style.backgroundColor = "darkorange";
        }            
}

function fontDemo(x) {
        if (x.style.fontFamily == "times new roman") {
            x.style.removeProperty('font-family');
        } else {
            x.style.fontFamily = "times new roman";
        }
}

function borderDemo(x) {
    if (x.className == "redborder") {
            x.classList.remove('redborder');
        } else {
            x.classList.add('redborder');
        }
}

function alignDemo(x) {
    if (x.style.textAlign == "left") {
            x.style.removeProperty('text-align');
        } else {
            x.style.textAlign = "left";
        }
}

function imageSizeDemo(x) {
    if (x.style.width == "200px") {
            x.style.removeProperty('width');
        } else {
            x.style.width = "200px";
        }
}

function resetDemo(text, box, image) {
    text.style.removeProperty('color');
    text.style.removeProperty('background-color'); 
    text.style.removeProperty('font-size');
    text.style.removeProperty('font-family');
    text.style.removeProperty('text-align');
    box.style.removeProperty('padding');
    box.style.removeProperty('background-color');
    box.classList.remove('redborder');
    image.style.removeProperty('width');
}
